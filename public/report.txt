<script >
    

const API_SALES = "http://localhost:3000/sales";
const API_CREDIT = "http://localhost:3000/creditsales";
const API_PROCUREMENT = "http://localhost:3000/procurement";

const Authorization = "Bearer " + localStorage.getItem("token");

// Elements
const totalSalesEl = document.querySelector(".cards .card:nth-child(1) h2");
const regularSalesEl = document.querySelector(".cards .card:nth-child(2) h2");
const creditSalesEl = document.querySelector(".cards .card:nth-child(3) h2");

const regularTableBody = document.querySelector("table tbody");
const stockTableBody = document.querySelectorAll("table tbody")[1];

let barChart;
let pieChart;

// ============================
// LOAD REPORT DATA
// ============================
async function loadReport() {
  try {

    const [salesRes, creditRes, procurementRes] = await Promise.all([
      fetch(API_SALES, { headers: { Authorization }}),
      fetch(API_CREDIT, { headers: { Authorization }}),
      fetch(API_PROCUREMENT, { headers: { Authorization }})
    ]);

    const salesData = (await salesRes.json()).data || [];
    const creditData = (await creditRes.json()).data || [];
    const procurementData = (await procurementRes.json()).data || [];

    // ============================
    // CALCULATIONS
    // ============================

    const totalRegular = salesData.reduce((sum, s) => sum + Number(s.amountPaid || 0), 0);
    const totalCredit = creditData.reduce((sum, s) => sum + Number(s.amountDue || 0), 0);
    const totalAll = totalRegular + totalCredit;

    totalSalesEl.innerText = "UGX " + totalAll.toLocaleString();
    regularSalesEl.innerText = "UGX " + totalRegular.toLocaleString();
    creditSalesEl.innerText = "UGX " + totalCredit.toLocaleString();

    // ============================
    // REGULAR SALES TABLE
    // ============================

    regularTableBody.innerHTML = "";

    salesData.forEach(s => {
      regularTableBody.innerHTML += `
        <tr>
          <td>${s.date} ${s.time}</td>
          <td>${s.produceName}</td>
          <td>-</td>
          <td>${s.tonnage} tons</td>
          <td>${s.buyerName}</td>
          <td>${s.salesAgentName}</td>
          <td>-</td>
          <td>UGX ${Number(s.amountPaid).toLocaleString()}</td>
        </tr>
      `;
    });

    // ============================
    // STOCK OVERVIEW
    // ============================

    stockTableBody.innerHTML = "";

    procurementData.forEach(p => {

      const soldTonnage = salesData
        .filter(s => s.produceName === p.nameOfProduce)
        .reduce((sum, s) => sum + Number(s.tonnage || 0), 0);

      const remaining = p.tonnage - soldTonnage;

      stockTableBody.innerHTML += `
        <tr>
          <td>${p.nameOfProduce}</td>
          <td>${p.typeOfProduce}</td>
          <td>${p.branch}</td>
          <td>${remaining} kg</td>
          <td>UGX ${p.sellingPrice.toLocaleString()}</td>
          <td>${p.dealerName}</td>
          <td>
            ${
              remaining <= 0
                ? `<span class="badge">Out of Stock</span>`
                : "Available"
            }
          </td>
        </tr>
      `;
    });

    // ============================
    // CHARTS
    // ============================

    if (barChart) barChart.destroy();
    if (pieChart) pieChart.destroy();

    barChart = new Chart(document.getElementById("barChart"), {
      type: "bar",
      data: {
        labels: ["Regular Sales", "Credit Sales"],
        datasets: [{
          label: "UGX",
          data: [totalRegular, totalCredit],
          backgroundColor: ["#1abc9c", "#f39c12"]
        }]
      }
    });

    pieChart = new Chart(document.getElementById("pieChart"), {
      type: "pie",
      data: {
        labels: ["Regular Sales", "Credit Sales"],
        datasets: [{
          data: [totalRegular, totalCredit],
          backgroundColor: ["#1abc9c", "#f39c12"]
        }]
      }
    });

  } catch (err) {
    console.log("Error loading report", err);
  }
}

loadReport();
</script>
